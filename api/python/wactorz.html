192    async def stop(self):
193        """Gracefully stop the actor."""
194        self.state = ActorState.STOPPED
195        for task in self._tasks:
196            task.cancel()
197        # Shield cleanup from CancelledError — chat tasks run as fire-and-forget
198        # asyncio tasks outside actor._tasks and get cancelled by asyncio.run()
199        # cleanup BEFORE these awaits if we don't shield them.
200        try:
201            await asyncio.shield(self.on_stop())
202        except (asyncio.CancelledError, Exception):
203            pass
204        try:
205            await asyncio.shield(self._save_persistent_state())
206        except (asyncio.CancelledError, Exception):
207            pass
208
209        # ── Persist message count so overview survives restarts ──────────
210        if self.metrics.messages_processed > 0:
211            self.persist("_messages_processed", {"count": self.metrics.messages_processed})
212
213        # ── Persist final cost metrics (for LLM-backed agents) ─────────
214        # Cost data lives in-memory and dies with the agent object.
215        # Persist it so the UI can show lifetime costs for deleted agents.
216        if hasattr(self, "total_cost_usd") and getattr(self, "total_cost_usd", 0) > 0:
217            self.persist("_final_cost", {
218                "input_tokens":  getattr(self, "total_input_tokens", 0),
219                "output_tokens": getattr(self, "total_output_tokens", 0),
220                "cost_usd":      round(self.total_cost_usd, 6),
221                "name":          self.name,
222                "stopped_at":    time.time(),
223            })
224
225        # ── Publish final metrics before the agent disappears ──────────
226        # The heartbeat loop is already cancelled at this point, so this
227        # is the UI's last chance to capture cost/usage data.
228        try:
229            final_metrics = self._build_metrics()
230            final_metrics["final"] = True   # signals UI this is the last message
231            await self._mqtt_publish(
232                f"agents/{self.actor_id}/metrics", final_metrics,
233            )
234        except Exception:
235            pass
236
237        await self._publish_status()
238        # ── Unregister from TopicBus ───────────────────────────────────
239        # Remove this agent's TopicContract so the planner doesn't wire
240        # against topics from stopped/deleted/replaced agents.
241        try:
242            from .topic_bus import get_topic_bus
243            bus = get_topic_bus()
244            if bus:
245                bus.unregister(self.name)
246        except Exception:

260    async def _message_loop(self):
261        """Main message processing loop."""
262        while self.state not in (ActorState.STOPPED, ActorState.FAILED):
263            try:
264                if self.state == ActorState.PAUSED:
265                    await asyncio.sleep(0.1)
266                    continue
267
268                msg = await asyncio.wait_for(self._mailbox.get(), timeout=1.0)
269                # Only count meaningful messages — not heartbeats, status pings, lifecycle
270                _noise = {MessageType.HEARTBEAT, MessageType.STATUS_REQUEST,
271                          MessageType.STATUS_RESPONSE, MessageType.STOP,
272                          MessageType.PAUSE, MessageType.RESUME}
273                if msg.type not in _noise:
274                    self.metrics.messages_processed += 1
275                await self._dispatch(msg)
276                self._mailbox.task_done()
277
278            except asyncio.TimeoutError:
279                continue
280            except asyncio.CancelledError:
281                break
282            except Exception as e:
283                self.metrics.errors += 1
284                logger.error(f"[{self.name}] Error in message loop: {e}", exc_info=True)

348    def _estimate_memory_mb(self) -> float:
349        """Bounded deep-size walk over this actor's own data structures.
350        Uses only sys.getsizeof — no new deps, works on Windows."""
351        seen: set = set()
352
353        def _deep(obj: object, depth: int) -> int:
354            if depth > 5:
355                return 0
356            oid = id(obj)
357            if oid in seen:
358                return 0
359            seen.add(oid)
360            sz = sys.getsizeof(obj)
361            if isinstance(obj, dict):
362                sz += sum(_deep(k, depth + 1) + _deep(v, depth + 1) for k, v in obj.items())
363            elif isinstance(obj, (list, tuple, set, frozenset)):
364                sz += sum(_deep(item, depth + 1) for item in obj)
365            return sz
366
367        total = 0
368        for attr in ("_persistent_state", "_outbox", "metrics"):
369            val = getattr(self, attr, None)
370            if val is not None:
371                total += _deep(val, 0)
372        # Rough proxy for messages sitting in the mailbox (~512 B each)
373        total += self._mailbox.qsize() * 512
374        return total / (1024 * 1024)

405    async def _command_listener(self):
406        """Listen for commands published to agents/{id}/commands via MQTT."""
407        try:
408            import aiomqtt
409        except ImportError:
410            return
411
412        topic = f"agents/{self.actor_id}/commands"
413        while self.state not in (ActorState.STOPPED, ActorState.FAILED):
414            try:
415                async with aiomqtt.Client(self._mqtt_broker, self._mqtt_port) as client:
416                    await client.subscribe(topic)
417                    logger.debug(f"[{self.name}] Subscribed to {topic}")
418                    async for message in client.messages:
419                        try:
420                            data    = json.loads(message.payload.decode())
421                            command = data.get("command", "")
422                            logger.info(f"[{self.name}] Received command: {command}")
423                            if self.protected and command in ("stop", "pause", "delete"):
424                                logger.warning(f"[{self.name}] Ignoring '{command}' — actor is protected.")
425                                continue
426                            if command == "stop":
427                                # ── Erlang unlink: release from supervision FIRST ──
428                                # If we just stop without releasing, the heartbeat-silence
429                                # watchdog will fire ~35s later and restart us anyway.
430                                if self._registry and hasattr(self._registry, "_supervisor_ref"):
431                                    sup = self._registry._supervisor_ref
432                                    if sup is not None:
433                                        sup.release(self.name)
434                                await self.stop()
435                                return
436                            elif command == "pause":
437                                await self.pause()
438                            elif command == "resume":
439                                await self.resume()
440                            elif command == "delete":
441                                # ── Erlang unlink: release from supervision FIRST ──
442                                if self._registry and hasattr(self._registry, "_supervisor_ref"):
443                                    sup = self._registry._supervisor_ref
444                                    if sup is not None:
445                                        sup.release(self.name)
446                                # If main actor knows about this agent, remove from spawn registry
447                                if self._registry:
448                                    main = self._registry.find_by_name("main")
449                                    if main and hasattr(main, "_remove_from_spawn_registry"):
450                                        main._remove_from_spawn_registry(self.name)
451                                    await self._registry.unregister(self.actor_id)
452                                await self.stop()
453                                return
454                        except Exception as e:
455                            logger.error(f"[{self.name}] Command parse error: {e}")
456            except asyncio.CancelledError:
457                break
458            except Exception as e:
459                if self.state not in (ActorState.STOPPED, ActorState.FAILED):
460                    await asyncio.sleep(5)

486    async def spawn(self, actor_class: type, **kwargs) -> "Actor":
487        """
488        Spawn a child actor. The child inherits:
489        - MQTT client (so it can publish heartbeats/status)
490        - Registry (so it can send/receive messages)
491        - Persistence dir defaults to same root
492        - Persistence API (SQLite/Redis/Pickle routing)
493
494        Erlang/OTP supervision: if the owning ActorSystem has a Supervisor,
495        the child is automatically registered under it with ONE_FOR_ONE so it
496        will be restarted if it crashes — no child is an orphan.
497        """
498        # Default persistence to same root as parent
499        kwargs.setdefault("persistence_dir", str(self._persistence_dir.parent))
500
501        child = actor_class(**kwargs)
502
503        # Inherit everything from parent
504        child._mqtt_client  = self._mqtt_client   # MQTT publish connection
505        child._mqtt_broker  = self._mqtt_broker   # broker address for command listener
506        child._mqtt_port    = self._mqtt_port     # broker port
507        child._registry     = self._registry      # message routing
508
509        # Inherit persistence API if available
510        if self._persistence_api is not None:
511            try:
512                from .persistence import PersistenceAPI, get_db, get_redis, get_pickle_store
513                db = get_db()
514                redis = get_redis()
515                pkl = get_pickle_store()
516                if db and redis and pkl:
517                    child._persistence_api = PersistenceAPI(db, redis, pkl, child.name)
518            except ImportError:
519                pass
520
521        # Register in registry
522        if self._registry:
523            await self._registry.register(child)
524
525        # Start the child
526        await child.start()
527
528        # ── Erlang/OTP: register child under Supervisor so it's never an orphan ──
529        # We reach into the registry to find the ActorSystem's supervisor.
530        # If no supervisor is available this is a safe no-op.
531        try:
532            if self._registry and hasattr(self._registry, "_supervisor_ref"):
533                supervisor = self._registry._supervisor_ref
534                if supervisor is not None and child.name not in supervisor._specs:
535                    # Capture child class + kwargs for the factory closure
536                    _child_class  = actor_class
537                    _child_kwargs = dict(kwargs)
538                    _child_name   = child.name
539                    _mqtt_client  = self._mqtt_client
540                    _mqtt_broker  = self._mqtt_broker
541                    _mqtt_port    = self._mqtt_port
542                    _persistence_api = self._persistence_api
543
544                    async def _child_factory(
545                        cls=_child_class, kw=_child_kwargs,
546                        mc=_mqtt_client, mb=_mqtt_broker, mp=_mqtt_port,
547                        papi=_persistence_api
548                    ):
549                        c = cls(**kw)
550                        c._mqtt_client = mc
551                        c._mqtt_broker = mb
552                        c._mqtt_port   = mp
553                        if papi is not None:
554                            try:
555                                from .persistence import PersistenceAPI, get_db, get_redis, get_pickle_store
556                                db = get_db(); redis = get_redis(); pkl = get_pickle_store()
557                                if db and redis and pkl:
558                                    c._persistence_api = PersistenceAPI(db, redis, pkl, c.name)
559                            except ImportError:
560                                pass
561                        return c
562
563                    supervisor.supervise(
564                        child.name,
565                        _child_factory,
566                        strategy   = SupervisorStrategy.ONE_FOR_ONE,
567                        max_restarts   = 5,
568                        restart_window = 60.0,
569                        restart_delay  = 2.0,
570                    )
571                    # Point spec.actor at the already-running child so the watch loop
572                    # starts monitoring immediately without a redundant restart.
573                    supervisor._specs[child.name].actor = child
574                    if child.name not in supervisor._order:
575                        supervisor._order.append(child.name)
576                    child.supervisor_id = id(supervisor)
577                    logger.info(
578                        f"[{self.name}] Child '{child.name}' auto-registered under Supervisor."
579                    )
580        except Exception as _sup_err:
581            # Never let supervision registration crash the spawn itself
582            logger.warning(
583                f"[{self.name}] Could not auto-supervise child '{child.name}': {_sup_err}"
584            )
585
586        # Immediately announce to monitor - don't wait for heartbeat loop
587        await child._publish_status()
588        await child._mqtt_publish(
589            f"agents/{child.actor_id}/heartbeat",
590            child._build_heartbeat(),
591        )
592        await child._mqtt_publish(
593            f"agents/{child.actor_id}/metrics",
594            child._build_metrics(),
595        )
596
597        # Notify parent's topic that it spawned a child
598        await self._mqtt_publish(
599            f"agents/{self.actor_id}/spawned",
600            {"child_id": child.actor_id, "child_name": child.name, "timestamp": time.time()},
601        )
602        logger.info(f"[{self.name}] Spawned: {child.name} ({child.actor_id[:8]})")
603        return child
